	transform = N2D2.DCTTransformation()

	stimuli = N2D2.StimuliProvider(database, [24, 24, 1], 1, False)
	N2D2.AnchorCell_Frame(deepNet, "name", stimuli, [N2D2.Anchor()], 1)

	N2D2.BatchNormCell_Frame_float(deepNet, "name", 10)
	N2D2.BatchNormCell_Frame_CUDA_float(deepNet, "name", 10)

	N2D2.ConvCell_Frame_CUDA_float(deepNet, "name", [5, 5], 24, [1, 1], [2, 2], [5, 5], [1, 1])

	N2D2.DeconvCell_Frame_CUDA_float(deepNet, "name", [5, 5], 24, [2, 2], [5, 5], [1, 1])
	N2D2.DropoutCell_Frame_float(deepNet, "name", 10)
	N2D2.DropoutCell_Frame_CUDA_float(deepNet, "name", 10)
	N2D2.ElemWiseCell_Frame(deepNet, "name", 10)
	N2D2.ElemWiseCell_Frame_CUDA(deepNet, "name", 10, N2D2.ElemWiseCell.Operation.Sum)
	N2D2.FcCell_Spike_Analog(net, deepNet, "name", 10)
	N2D2.FcCell_Spike_PCM(net, deepNet, "name", 10)
	N2D2.FcCell_Spike_RRAM(net, deepNet, "name", 10)
	N2D2.FMPCell_Frame(deepNet, "name", 2, 10)
	N2D2.FMPCell_Frame_CUDA(deepNet, "name", 2, 10)
	N2D2.LRNCell_Frame_float(deepNet, "name", 10)
	N2D2.LRNCell_Frame_CUDA_float(deepNet, "name", 10)

	N2D2.NormalizeCell_Frame_float(deepNet, "name", 10, N2D2.NormalizeCell.Norm.L1)
	N2D2.NormalizeCell_Frame_CUDA_float(deepNet, "name", 10, N2D2.NormalizeCell.Norm.L1)
	N2D2.ObjectDetCell_Frame(deepNet, "name", stimuli, 1, 1, 1, 1)
	N2D2.ObjectDetCell_Frame_CUDA(deepNet, "name", stimuli, 1, 1, 1, 1)
